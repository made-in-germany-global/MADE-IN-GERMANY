// Language toggle functionality
document.addEventListener('DOMContentLoaded', function() {
  // Get language buttons and content divs
  const enButton = document.getElementById('en-button');
  const deButton = document.getElementById('de-button');
  const enContent = document.getElementById('english');
  const deContent = document.getElementById('german');
  
  // English button click
  enButton.addEventListener('click', function(e) {
    e.preventDefault();
    enContent.style.display = 'block';
    deContent.style.display = 'none';
    enButton.style.backgroundColor = '#FFD700';
    enButton.style.color = '#000';
    deButton.style.backgroundColor = '#333';
    deButton.style.color = '#fff';
  });
  
  // German button click
  deButton.addEventListener('click', function(e) {
    e.preventDefault();
    enContent.style.display = 'none';
    deContent.style.display = 'block';
    deButton.style.backgroundColor = '#FFD700';
    deButton.style.color = '#000';
    enButton.style.backgroundColor = '#333';
    enButton.style.color = '#fff';
  });
  
  // Add animation effects
  const industryItems = document.querySelectorAll('.industry-item');
  
  // Stagger animation on load
  industryItems.forEach((item, index) => {
    setTimeout(() => {
      item.style.opacity = '1';
      item.style.transform = 'translateY(0)';
    }, 100 * index);
  });
  
  // Initialize all items with opacity 0 and transform
  industryItems.forEach(item => {
    item.style.opacity = '0';
    item.style.transform = 'translateY(20px)';
    item.style.transition = 'opacity 0.5s ease, transform 0.5s ease';
  });
});
</script>
